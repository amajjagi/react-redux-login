import { FETCH_POSTS, NEW_POST}  from './types'



/**
 * fETCH POST WILL GIVE ALL THE POSTS
 */
export const fetchPosts = () => dispatch =>{
    fetch('https://jsonplaceholder.typicode.com/posts')
    .then(res => res.json())
    .then(posts => {
        dispatch ({
            type: FETCH_POSTS,
            payload: posts
        })
    });
}
